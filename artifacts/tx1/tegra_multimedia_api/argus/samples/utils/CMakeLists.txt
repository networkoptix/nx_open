cmake_minimum_required (VERSION 2.6)

project(argussampleutils)

set(CMAKE_MODULE_PATH "${CMAKE_SOURCE_DIR}/cmake" "${CMAKE_MODULE_PATH}")

find_package(Argus REQUIRED)
find_package(OpenGLES REQUIRED)
find_package(EGL REQUIRED)
find_package(X11 REQUIRED)
find_package(CUDA)

find_package(PkgConfig REQUIRED)
pkg_check_modules(GTK3 REQUIRED gtk+-3.0)

set(SOURCES
    EGLGlobal.cpp
    GLContext.cpp
    JPEGConsumer.cpp
    Observed.cpp
    PreviewConsumer.cpp
    Thread.cpp
    WindowBase.cpp
    gtk/GuiElement.cpp
    gtk/Window.cpp
    )

if(CUDA_FOUND)
    set(SOURCES
        ${SOURCES}
        CUDAHelper.cpp
        )
endif(CUDA_FOUND)

include_directories(
    ${ARGUS_INCLUDE_DIR}
    ${EGL_INCLUDE_DIR}
    ${OPENGLES_INCLUDE_DIR}
    ${GTK3_INCLUDE_DIRS}
    ${CMAKE_CURRENT_SOURCE_DIR}
    )

link_directories(
    ${GTK3_LIBRARY_DIRS}
    )

add_definitions(
    ${GTK3_CFLAGS_OTHER}
    )

if(CUDA_FOUND)
    cuda_add_library(${PROJECT_NAME} ${SOURCES})
    target_link_libraries(${PROJECT_NAME}
        ${CUDA_LIBRARIES} )
else(CUDA_FOUND)
    add_library(${PROJECT_NAME} STATIC ${SOURCES})
endif(CUDA_FOUND)

target_link_libraries(${PROJECT_NAME}
    pthread
    ${X11_LIBRARIES}
    ${EGL_LIBRARIES}
    ${OPENGLES_LIBRARIES}
    ${GTK3_LIBRARIES}
    )
